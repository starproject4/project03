//회원 테이블
CREATE TABLE  "MEMBER" 
   (	"ID" VARCHAR2(30) NOT NULL ENABLE, 
	"PW" VARCHAR2(30) NOT NULL ENABLE, 
	"NAME" VARCHAR2(30) NOT NULL ENABLE, 
	"BIRTH" VARCHAR2(30) NOT NULL ENABLE, 
	"PHONE" VARCHAR2(30) NOT NULL ENABLE, 
	"EMAIL" VARCHAR2(30) NOT NULL ENABLE, 
	"ADDR" VARCHAR2(100) NOT NULL ENABLE, 
	"STAR" NUMBER(2,0) NOT NULL ENABLE, 
	"GRADE" VARCHAR2(30) NOT NULL ENABLE, 
	"POINT" NUMBER(5,0), 
	 CONSTRAINT "MEMBER_PK" PRIMARY KEY ("ID") ENABLE, 
	 CONSTRAINT "PW_CK1" CHECK (LENGTH(PW)>=6) ENABLE
   );
   
   
   //메뉴 테이블 수정
   CREATE TABLE  "MENU" 
   (	"NO" NUMBER, 
	"TITLE" VARCHAR2(100) NOT NULL ENABLE, 
	"CONTENT" VARCHAR2(4000) NOT NULL ENABLE, 
	"KCAL" NUMBER, 
	"SODIUM" NUMBER, 
	"FAT" NUMBER, 
	"SACCHARIDE" NUMBER, 
	"PROTEIN" NUMBER, 
	"CAFFEINE" NUMBER, 
	"ORIGINALFILENAME" VARCHAR2(100), 
	"NEWFILENAME" VARCHAR2(100), 
	"NALJA" DATE DEFAULT sysdate NOT NULL ENABLE, 
	"CATEGORY" VARCHAR2(20) NOT NULL ENABLE, 
	"PRICE" NUMBER(5,0) NOT NULL ENABLE, 
	 PRIMARY KEY ("NO") ENABLE
   )
